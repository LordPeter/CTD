## Initial event to chose spawn type

# AI gets all the options at once
character_event = {
	id = CTD.0501
	
	is_triggered_only = yes		# by decision
	
	hide_window = yes	# no need to show for ai
	
	immediate = {
		random_list = {
			# hunting:
			10 = {
				modifier = {
					factor = 0
					hunting_spawn_possible_CTD_trigger = no
				}
				event_target:target_spawn_province = {
					add_province_modifier = {
						name = spawn_hunting
						duration = -1
					}
					# This is so the follow-up event knows which of the spawns is "new":
					set_province_flag = chosen_spawn_is_hunting
				}
			}
		}
		character_event = {	# Event to apply variables & decide on 1 or 2 products
			id = CTD.0505
		}
	}
}
