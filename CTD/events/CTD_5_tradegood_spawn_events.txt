## Events to decide on 1 or 2 spawned products & apply variables depending on both province cap and workforce

# First/Second product to spawn (can be re-called once)
character_event = {
	id = CTD.0505
	desc = CTDDESC_0505
	
	is_triggered_only = yes		# by event
	
	## Options for all the products
	
	# gold_ore:
	option = {
		name = CTDOPTA_0505
		trigger = {
			event_target:target_spawn_province = {
				AND = {
					NOT = {	# Not if just spawned the same product (for 2 product choice)
						has_province_flag = newly_spawned_gold_ore
					}
					OR = {	# list of spawns that can spawn this product
						has_province_flag = chosen_spawn_is_open_pit_mining
						has_province_flag = chosen_spawn_is_underground_mining
						has_province_flag = chosen_spawn_is_placer_mining
					}
					# for minerals: province has unused but known deposit:
					has_province_modifier = prospected_mineral_gold_ore
				}
			}
		}
		event_target:target_spawn_province = {
			set_province_flag = newly_spawned_gold_ore
		}
		# For Minerals, also remove the "prospected" modifer:
		remove_province_modifier = prospected_mineral_gold_ore
	}
	### add the rest NOTE
	
	after = {
		# Ask if want a 2nd product to spawn, if spawn can spawn a 2nd product and hasn't already spawned 2nd.
		if = {
			limit = {
				AND = {
					has_chosen_2_products_CTD_trigger = no	# NOTE code this!
					chosen_spawn_allows_for_2nd_product_CTD_trigger = yes	# NOTE code this!
				}
			}
			character_event = {
				id = CTD.0506
			}
		}
		# Otherwise wrap things up by firing the event that applies variables & stuff and matches to cap/workforce:
		else = {
			hidden_tooltip = {
				character_event = {
					id = CTD.0507
				}
			}
		}
	}
}
